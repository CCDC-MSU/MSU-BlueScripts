# CCDC Target Hosts with Credentials
# Supported formats:
# 1. host:user:password            (password authentication)
# 2. host:user:keyfile             (SSH key authentication)
# 3. host:user:password:port       (custom port)
# 4. host:user:password:name       (friendly name - starts with letter)
# 5. host:user:password:port:name  (port + friendly name)
#
# Friendly names make logs easier to read - use them!
# Lines starting with # are ignored

# Examples:
# 192.168.1.11:root:changeme123              # Password auth
# 192.168.1.12:admin:newpass:webserver       # With friendly name
# 192.168.1.13:root:~/.ssh/id_rsa            # SSH key auth
# 192.168.1.14:root:mypass:2222              # Custom SSH port
# 192.168.1.15:root:mypass:2222:database     # Port + friendly name
